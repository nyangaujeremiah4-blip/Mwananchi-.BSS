<?php
session_start();

// Database Configuration
define('DB_HOST', 'localhost');
define('DB_USER', 'root');
define('DB_PASS', '');
define('DB_NAME', 'mbss_db');

// Create connection
$conn = new mysqli(DB_HOST, DB_USER, DB_PASS, DB_NAME);

if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}

$conn->set_charset("utf8");

$logged_in = isset($_SESSION['user_id']);
$username = $_SESSION['username'] ?? '';
$user_id = $_SESSION['user_id'] ?? '';
$role = $_SESSION['role'] ?? '';

// Handle AJAX requests
header('Content-Type: application/json');

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $action = $_POST['action'] ?? '';

    // LOGIN
    if ($action == 'login') {
        $username_login = $_POST['username'] ?? '';
        $password_login = $_POST['password'] ?? '';

        $stmt = $conn->prepare("SELECT id, username, password, role FROM users WHERE username = ?");
        $stmt->bind_param("s", $username_login);
        $stmt->execute();
        $result = $stmt->get_result();

        if ($result->num_rows == 1) {
            $user = $result->fetch_assoc();
            if (password_verify($password_login, $user['password'])) {
                $_SESSION['user_id'] = $user['id'];
                $_SESSION['username'] = $user['username'];
                $_SESSION['role'] = $user['role'];
                echo json_encode(['success' => true, 'message' => 'Login successful']);
            } else {
                echo json_encode(['success' => false, 'message' => 'Invalid password']);
            }
        } else {
            echo json_encode(['success' => false, 'message' => 'User not found']);
        }
        exit;
    }

    // REGISTER
    if ($action == 'register') {
        $username_reg = $_POST['username'] ?? '';
        $email = $_POST['email'] ?? '';
        $password = $_POST['password'] ?? '';
        $role_reg = $_POST['role'] ?? '';
        $phone = $_POST['phone'] ?? '';
        $location = $_POST['location'] ?? '';

        if (empty($username_reg) || empty($email) || empty($password) || empty($role_reg)) {
            echo json_encode(['success' => false, 'message' => 'All required fields must be filled']);
            exit;
        }

        $stmt = $conn->prepare("SELECT id FROM users WHERE username = ? OR email = ?");
        $stmt->bind_param("ss", $username_reg, $email);
        $stmt->execute();
        $stmt->store_result();

        if ($stmt->num_rows > 0) {
            echo json_encode(['success' => false, 'message' => 'Username or email already exists']);
            exit;
        }

        $hashed_password = password_hash($password, PASSWORD_BCRYPT);
        $stmt = $conn->prepare("INSERT INTO users (username, email, password, role, phone_number, location) VALUES (?, ?, ?, ?, ?, ?)");
        $stmt->bind_param("ssssss", $username_reg, $email, $hashed_password, $role_reg, $phone, $location);

        if ($stmt->execute()) {
            echo json_encode(['success' => true, 'message' => 'Account created successfully! Please login.']);
        } else {
            echo json_encode(['success' => false, 'message' => 'Error creating account']);
        }
        exit;
    }

    // LOGOUT
    if ($action == 'logout') {
        session_destroy();
        echo json_encode(['success' => true, 'message' => 'Logged out']);
        exit;
    }

    // ADD PRODUCT (Seller)
    if ($action == 'add_product') {
        $product_name = $_POST['productName'] ?? '';
        $description = $_POST['description'] ?? '';
        $category = $_POST['category'] ?? '';
        $price = $_POST['price'] ?? 0;

        if (empty($product_name) || empty($description) || empty($category) || $price <= 0) {
            echo json_encode(['success' => false, 'message' => 'All fields are required']);
            exit;
        }

        $stmt = $conn->prepare("INSERT INTO products (seller_id, product_name, description, category, price, availability) VALUES (?, ?, ?, ?, ?, 'available')");
        $stmt->bind_param("isssd", $user_id, $product_name, $description, $category, $price);

        if ($stmt->execute()) {
            echo json_encode(['success' => true, 'message' => 'Product added successfully']);
        } else {
            echo json_encode(['success' => false, 'message' => 'Error adding product']);
        }
        exit;
    }

    // SEND MESSAGE
    if ($action == 'send_message') {
        $buyer_id = $_POST['buyer_id'] ?? 0;
        $seller_id = $_POST['seller_id'] ?? 0;
        $product_id = $_POST['product_id'] ?? 0;
        $sender_id = $_POST['sender_id'] ?? 0;
        $message = $_POST['message'] ?? '';

        $stmt = $conn->prepare("INSERT INTO chats (buyer_id, seller_id, product_id, sender_id, message, message_type) VALUES (?, ?, ?, ?, ?, 'text')");
        $stmt->bind_param("iiiis", $buyer_id, $seller_id, $product_id, $sender_id, $message);

        if ($stmt->execute()) {
            echo json_encode(['success' => true, 'message' => 'Message sent']);
        } else {
            echo json_encode(['success' => false, 'message' => 'Error sending message']);
        }
        exit;
    }

    // AGREE TO MEET
    if ($action == 'agree_meet') {
        $buyer_id = $_POST['buyer_id'] ?? 0;
        $seller_id = $_POST['seller_id'] ?? 0;
        $product_id = $_POST['product_id'] ?? 0;

        $stmt = $conn->prepare("UPDATE chats SET meet_agreed = 1, message_type = 'agreement' WHERE buyer_id = ? AND seller_id = ? AND product_id = ? LIMIT 1");
        $stmt->bind_param("iii", $buyer_id, $seller_id, $product_id);

        if ($stmt->execute()) {
            echo json_encode(['success' => true, 'message' => 'Meet agreement recorded']);
        } else {
            echo json_encode(['success' => false, 'message' => 'Error recording agreement']);
        }
        exit;
    }
}

// Handle GET requests
if ($_SERVER['REQUEST_METHOD'] == 'GET') {
    $action = $_GET['action'] ?? '';

    // SEARCH PRODUCTS
    if ($action == 'search') {
        $keyword = $_GET['keyword'] ?? '';
        $category = $_GET['category'] ?? '';
        $limit = $_GET['limit'] ?? 20;

        $keyword = strtolower(trim($keyword));
        $keyword_param = "%$keyword%";

        $query = "
            SELECT 
                p.id, p.product_name, p.description, p.category, p.price, p.availability,
                u.id AS seller_id, u.username AS seller_name, u.phone_number, u.location,
                CASE 
                    WHEN LOWER(p.product_name) LIKE LOWER(?) THEN 3
                    WHEN LOWER(p.description) LIKE LOWER(?) THEN 2
                    WHEN LOWER(p.category) LIKE LOWER(?) THEN 1
                    ELSE 0
                END AS relevance
            FROM products p
            JOIN users u ON p.seller_id = u.id
            WHERE p.availability = 'available' AND
                (LOWER(p.product_name) LIKE LOWER(?) OR 
                 LOWER(p.description) LIKE LOWER(?) OR 
                 LOWER(p.category) LIKE LOWER(?))
        ";

        if (!empty($category)) {
            $query .= " AND LOWER(p.category) = LOWER(?)";
        }

        $query .= " ORDER BY relevance DESC LIMIT ?";

        $stmt = $conn->prepare($query);

        if (!empty($category)) {
            $stmt->bind_param("ssssssi", $keyword_param, $keyword_param, $keyword_param, 
                             $keyword_param, $keyword_param, $keyword_param, $category, $limit);
        } else {
            $stmt->bind_param("sssssi", $keyword_param, $keyword_param, $keyword_param, 
                             $keyword_param, $keyword_param, $keyword_param, $limit);
        }

        $stmt->execute();
        $result = $stmt->get_result();
        $products = [];
        while ($row = $result->fetch_assoc()) {
            $products[] = $row;
        }
        echo json_encode(['success' => true, 'data' => $products]);
        exit;
    }

    // GET ALL PRODUCTS
    if ($action == 'all') {
        $limit = isset($_GET['limit']) ? (int)$_GET['limit'] : 20;
        $offset = isset($_GET['offset']) ? (int)$_GET['offset'] : 0;

        $stmt = $conn->prepare("
            SELECT p.id, p.product_name, p.description, p.category, p.price,
                   u.id AS seller_id, u.username AS seller_name, u.phone_number, u.location
            FROM products p
            JOIN users u ON p.seller_id = u.id
            WHERE p.availability = 'available'
            ORDER BY p.created_at DESC
            LIMIT ? OFFSET ?
        ");

        $stmt->bind_param("ii", $limit, $offset);
        $stmt->execute();
        $result = $stmt->get_result();
        $products = [];
        while ($row = $result->fetch_assoc()) {
            $products[] = $row;
        }
        echo json_encode(['success' => true, 'data' => $products]);
        exit;
    }

    // GET CATEGORIES
    if ($action == 'categories') {
        $result = $conn->query("SELECT DISTINCT category FROM products WHERE category IS NOT NULL ORDER BY category ASC");
        $categories = [];
        while ($row = $result->fetch_assoc()) {
            $categories[] = $row['category'];
        }
        echo json_encode(['success' => true, 'data' => $categories]);
        exit;
    }

    // GET CHAT HISTORY
    if ($action == 'chat_history') {
        $buyer_id = $_GET['buyer_id'] ?? 0;
        $seller_id = $_GET['seller_id'] ?? 0;
        $product_id = $_GET['product_id'] ?? 0;

        $stmt = $conn->prepare("
            SELECT c.id, c.message, c.sender_id, u.username, c.created_at
            FROM chats c
            JOIN users u ON c.sender_id = u.id
            WHERE c.product_id = ? AND 
                  ((c.buyer_id = ? AND c.seller_id = ?) OR (c.buyer_id = ? AND c.seller_id = ?))
            ORDER BY c.created_at ASC
        ");

        $stmt->bind_param("iiiii", $product_id, $buyer_id, $seller_id, $seller_id, $buyer_id);
        $stmt->execute();
        $result = $stmt->get_result();
        $messages = [];
        while ($row = $result->fetch_assoc()) {
            $messages[] = $row;
        }
        echo json_encode(['success' => true, 'data' => $messages]);
        exit;
    }

    // GET SELLER PRODUCTS
    if ($action == 'seller_products') {
        $seller_id = $_GET['seller_id'] ?? 0;

        $stmt = $conn->prepare("
            SELECT id, product_name, description, category, price, availability
            FROM products
            WHERE seller_id = ?
            ORDER BY created_at DESC
        ");

        $stmt->bind_param("i", $seller_id);
        $stmt->execute();
        $result = $stmt->get_result();
        $products = [];
        while ($row = $result->fetch_assoc()) {
            $products[] = $row;
        }
        echo json_encode(['success' => true, 'data' => $products]);
        exit;
    }
}

        // THEME MANAGEMENT
        function toggleTheme() {
            const body = document.body;
            const themes = ['light-theme', 'dark-theme', 'neon-theme'];
            let currentIndex = -1;

            themes.forEach((theme, index) => {
                if (body.classList.contains(theme)) {
                    currentIndex = index;
                }
            });

            body.classList.remove(...themes);
            const nextIndex = (currentIndex + 1) % themes.length;
            body.classList.add(themes[nextIndex]);
            localStorage.setItem('theme', themes[nextIndex]);
        }

        // AUTH HANDLERS
        function handleLogin(e) {
            e.preventDefault();
            
            const username = document.getElementById('loginUsername').value;
            const password = document.getElementById('loginPassword').value;

            const formData = new FormData();
            formData.append('action', 'login');
            formData.append('username', username);
            formData.append('password', password);

            fetch(window.location.pathname, {
                method: 'POST',
                body: formData
            })
            .then(res => res.json())
            .then(data => {
                const errDiv = document.getElementById('loginError');
                const sucDiv = document.getElementById('loginSuccess');
                
                if (data.success) {
                    sucDiv.textContent = data.message;
                    sucDiv.style.display = 'block';
                    setTimeout(() => location.reload(), 1500);
                } else {
                    errDiv.textContent = data.message;
                    errDiv.style.display = 'block';
                }
            })
            .catch(err => {
                document.getElementById('loginError').textContent = 'Error: ' + err.message;
                document.getElementById('loginError').style.display = 'block';
                console.error(err);
            });
        }

        function performSearch() {
            const keyword = document.getElementById('searchInput').value;
            const category = document.getElementById('categoryFilter').value;

            if (!keyword.trim()) {
                alert('Enter a search term');
                return;
            }

            fetch(window.location.pathname + `?action=search&keyword=${encodeURIComponent(keyword)}&category=${encodeURIComponent(category)}`)
                .then(res => res.json())
                .then(data => {
                    if (data.success) {
                        displayProducts(data.data);
                    }
                });
        }
